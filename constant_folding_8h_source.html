177class ConstantFolding : public PassManager {
178 public:
179 ConstantFolding(ReferenceMap *refMap, TypeMap *typeMap, ConstantFoldingPolicy *policy)
180 : ConstantFolding(refMap, typeMap, true, nullptr, policy) {}
181
182 ConstantFolding(ReferenceMap *refMap, TypeMap *typeMap, bool warnings = true,
183 TypeChecking *typeChecking = nullptr, ConstantFoldingPolicy *policy = nullptr) {
184 if (typeMap != nullptr) {
185 if (!typeChecking) typeChecking = new TypeChecking(refMap, typeMap);
186 passes.push_back(typeChecking);
187 }
188 passes.push_back(new DoConstantFolding(refMap, typeMap, warnings, policy));
189 if (typeMap != nullptr) passes.push_back(new ClearTypeMap(typeMap));
190 setName("ConstantFolding");
191 }
192};
