43// declare this outside of Visitor so it can be forward declared in node.h
44struct Visitor_Context {
45 // We maintain a linked list of Context structures on the stack
46 // in the Visitor::apply_visitor functions as we do the recursive
47 // descent traversal. pre/postorder function can access this
48 // context via getContext/findContext
49 const Visitor_Context *parent;
50 const IR::Node *node, *original;
51 mutable const char *child_name;
52 mutable int child_index;
53 int depth;
54 template <class T>
55 inline const T *findContext(const Visitor_Context *&c) const {
56 c = this;
57 while ((c = c->parent))
58 if (auto *rv = c->node->to<T>()) return rv;
59 return nullptr;
60 }
61 template <class T>
62 inline const T *findContext() const {
63 const Visitor_Context *c = this;
64 return findContext<T>(c);
65 }
66};

608class SplitFlowVisit_base {
609 protected:
610 Visitor &v;
611 SplitFlowVisit_base *prev;
612 std::vector<Visitor *> visitors;
613 int visit_next = 0, start_index = 0;
614 bool paused = false;
615 friend ControlFlowVisitor;
616
617 explicit SplitFlowVisit_base(Visitor &v) : v(v) {
618 prev = v.split_link;
619 v.split_link = this;
620 }
621 ~SplitFlowVisit_base() { v.split_link = prev; }
622 void *operator new(size_t); // declared and not defined, as this class can
623 // only be instantiated on the stack. Trying to allocate one on the heap will
624 // cause a linker error.
625
626 public:
627 bool finished() { return size_t(visit_next) >= visitors.size(); }
628 virtual bool ready() { return !finished() && !paused; }
629 void pause() { paused = true; }
630 void unpause() { paused = false; }
631 virtual void do_visit() = 0;
632 virtual void run_visit() {
633 auto *ctxt = v.getChildContext();
634 start_index = ctxt ? ctxt->child_index : 0;
635 paused = false;
636 while (!finished()) do_visit();
637 for (auto *cl : visitors) {
638 if (cl && cl != &v) v.flow_merge(*cl);
639 }
640 }
641 virtual void dbprint(std::ostream &) const = 0;
642 friend void dump(const SplitFlowVisit_base *);
643};

662 BUG_CHECK(visit_next == 0, "Can't addNode to SplitFlowVisit after visiting started");

691class SplitFlowVisitVector : public SplitFlowVisit_base {
692 IR::Vector<N> *vec = nullptr;
693 const IR::Vector<N> *const_vec = nullptr;
694 std::vector<const IR::Node *> result;
695 void init_visit(size_t size) {
696 if (size > 0) visitors.push_back(&v);
697 while (visitors.size() < size) visitors.push_back(&v.flow_clone());
698 }
699
700 public:
701 SplitFlowVisitVector(Visitor &v, IR::Vector<N> &vec) : SplitFlowVisit_base(v), vec(&vec) {
702 result.resize(vec.size());
703 init_visit(vec.size());
704 }
705 SplitFlowVisitVector(Visitor &v, const IR::Vector<N> &vec)
706 : SplitFlowVisit_base(v), const_vec(&vec) {
707 init_visit(vec.size());
708 }
709 void do_visit() override {
710 if (!finished()) {
711 BUG_CHECK(!paused, "trying to visit paused split_flow_visitor");
712 int idx = visit_next++;
713 if (vec)
714 result[idx] = visitors.at(idx)->apply_visitor(vec->at(idx));
715 else
716 visitors.at(idx)->visit(const_vec->at(idx), nullptr, start_index + idx);
717 }
718 }
719 void run_visit() override {
720 SplitFlowVisit_base::run_visit();
721 if (vec) {
722 int idx = 0;
723 for (auto i = vec->begin(); i != vec->end(); ++idx) {
724 if (!result[idx] && *i) {
725 i = vec->erase(i);
726 } else if (result[idx] == *i) {
727 ++i;
728 } else if (auto l = result[idx]->template to<IR::Vector<N>>()) {
729 i = vec->erase(i);
730 i = vec->insert(i, l->begin(), l->end());
731 i += l->size();
732 } else if (auto v = result[idx]->template to<IR::VectorBase>()) {
733 if (v->empty()) {
734 i = vec->erase(i);
735 } else {
736 i = vec->insert(i, v->size() - 1, nullptr);
737 for (auto el : *v) {
738 CHECK_NULL(el);
739 if (auto e = el->template to<N>())
740 *i++ = e;
741 else
742 BUG("visitor returned invalid type %s for Vector<%s>",
743 el->node_type_name(), N::static_type_name());
744 }
745 }
746 } else if (auto e = result[idx]->template to<N>()) {
747 *i++ = e;
748 } else {
749 CHECK_NULL(result[idx]);
750 BUG("visitor returned invalid type %s for Vector<%s>",
751 result[idx]->node_type_name(), N::static_type_name());
752 }
753 }
754 }
755 }
756 void dbprint(std::ostream &out) const override {
757 out << "SplitFlowVisitVector processed " << visit_next << " of " << visitors.size();
