30class HasTableApply : public Inspector {
31 ReferenceMap *refMap;
32 TypeMap *typeMap;
33
34 public:
35 const IR::P4Table *table;
36 const IR::MethodCallExpression *call;
37 HasTableApply(ReferenceMap *refMap, TypeMap *typeMap)
38 : refMap(refMap), typeMap(typeMap), table(nullptr), call(nullptr) {
39 CHECK_NULL(refMap);
40 CHECK_NULL(typeMap);
41 setName("HasTableApply");
42 }
43
44 void postorder(const IR::MethodCallExpression *expression) override {
45 auto mi = MethodInstance::resolve(expression, refMap, typeMap);
46 if (!mi->isApply()) return;
47 auto am = mi->to<P4::ApplyMethod>();
48 if (!am->object->is<IR::P4Table>()) return;
49 BUG_CHECK(table == nullptr, "%1% and %2%: multiple table applications in one expression",
50 table, am->object);
51 table = am->object->to<IR::P4Table>();
52 call = expression;
53 LOG3("Invoked table is " << dbp(table));
54 }
55};

64class SideEffects : public Inspector {
65 private:
66 ReferenceMap *refMap;
67 TypeMap *typeMap;
68
69 public:
71 const IR::Node *nodeWithSideEffect = nullptr;
72
74 unsigned sideEffectCount = 0;
75
76 void postorder(const IR::MethodCallExpression *mce) override {
77 if (refMap == nullptr || typeMap == nullptr) {
78 // conservative
79 sideEffectCount++;
80 nodeWithSideEffect = mce;
81 return;
82 }
83 auto mi = MethodInstance::resolve(mce, refMap, typeMap);
84 if (!mi->is<BuiltInMethod>()) {
85 sideEffectCount++;
86 nodeWithSideEffect = mce;
87 return;
88 }
89 auto bim = mi->to<BuiltInMethod>();
90 if (bim->name.name != IR::Type_Header::isValid) {
91 sideEffectCount++;
92 nodeWithSideEffect = mce;
93 }
94 }
95
96 void postorder(const IR::ConstructorCallExpression *cce) override {
97 sideEffectCount++;
98 nodeWithSideEffect = cce;
99 }
100
103 SideEffects(ReferenceMap *refMap, TypeMap *typeMap) : refMap(refMap), typeMap(typeMap) {
104 setName("SideEffects");
105 }
106
108 static bool check(const IR::Expression *expression, const Visitor *calledBy,
109 ReferenceMap *refMap, TypeMap *typeMap) {
110 SideEffects se(refMap, typeMap);
111 se.setCalledBy(calledBy);
112 expression->apply(se);
113 return se.nodeWithSideEffect != nullptr;
114 }
116 static bool hasSideEffect(const IR::MethodCallExpression *mce, ReferenceMap *refMap,
117 TypeMap *typeMap) {
118 // mce does not produce a side effect in few cases:
119 // * isValid()
120 // * function, extern function, or extern method with noSideEffects annotation
121 auto mi = MethodInstance::resolve(mce, refMap, typeMap);
122 if (auto em = mi->to<P4::ExternMethod>()) {
123 if (em->method->getAnnotation(IR::Annotation::noSideEffectsAnnotation)) {
124 return false;
125 }
126 return true;
127 }
128 if (auto ef = mi->to<P4::ExternFunction>()) {
129 if (ef->method->getAnnotation(IR::Annotation::noSideEffectsAnnotation)) {
130 return false;
131 }
132 return true;
133 }
134 if (auto ef = mi->to<P4::FunctionCall>()) {
135 if (ef->function->getAnnotation(IR::Annotation::noSideEffectsAnnotation)) {
136 return false;
137 }
138 return true;
139 }
140 if (auto bim = mi->to<BuiltInMethod>()) {
141 if (bim->name.name == IR::Type_Header::isValid) {
142 return false;
143 }
144 }
145 return true;
146 }
147};

341class TablesInKeys : public Inspector {
342 ReferenceMap *refMap;
343 TypeMap *typeMap;
344 std::set<const IR::P4Table *> *invokedInKey;
345
346 public:
347 TablesInKeys(ReferenceMap *refMap, TypeMap *typeMap,
348 std::set<const IR::P4Table *> *invokedInKey)
349 : refMap(refMap), typeMap(typeMap), invokedInKey(invokedInKey) {
350 CHECK_NULL(invokedInKey);
351 setName("TableInKeys");
352 }
353 Visitor::profile_t init_apply(const IR::Node *node) override {
354 invokedInKey->clear();
355 return Inspector::init_apply(node);
356 }
357 void postorder(const IR::MethodCallExpression *mce) override {
358 if (!findContext<IR::Key>()) return;
359 HasTableApply hta(refMap, typeMap);
360 hta.setCalledBy(this);
361 (void)mce->apply(hta);
362 if (hta.table != nullptr) {
363 LOG2("Table " << hta.table << " invoked in key of another table");
364 invokedInKey->emplace(hta.table);
365 }
366 }
367};

393class SideEffectOrdering : public PassRepeated {
394 // Contains all tables that are invoked within key
395 // computations for other tables. The keys for these
396 // inner tables cannot be expanded until the keys of
397 // the caller tables have been expanded.
398 std::set<const IR::P4Table *> invokedInKey;
399 // Temporaries that were added
400 std::set<const IR::Expression *> added;
401
402 public:
403 SideEffectOrdering(ReferenceMap *refMap, TypeMap *typeMap, bool skipSideEffectOrdering,
404 TypeChecking *typeChecking = nullptr) {
405 if (!typeChecking) typeChecking = new TypeChecking(refMap, typeMap);
406 if (!skipSideEffectOrdering) {
407 passes.push_back(new TypeChecking(refMap, typeMap));
408 passes.push_back(new DoSimplifyExpressions(refMap, typeMap, &added));
409 passes.push_back(typeChecking);
410 passes.push_back(new TablesInActions(refMap, typeMap));
411 passes.push_back(new TablesInKeys(refMap, typeMap, &invokedInKey));
412 passes.push_back(new KeySideEffect(refMap, typeMap, &invokedInKey));
413 }
414 setName("SideEffectOrdering");
415 }
416};
