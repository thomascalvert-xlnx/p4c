54 // a location in the program. Includes the context from the visitor, which needs to
55 // be copied out of the Visitor::Context objects, as they are allocated on the stack and
56 // will become invalid as the IR traversal continues
57 struct loc_t {
58 const IR::Node *node;
59 const loc_t *parent;
60 bool operator<(const loc_t &a) const {
61 if (node != a.node) return node->id < a.node->id;
62 if (!a.parent) return parent != nullptr;
63 return *parent < *a.parent;
64 }
65 template <class T>
66 const T *find() const {
67 for (auto *p = this; p; p = p->parent) {
68 if (auto *t = p->node->to<T>()) return t;
69 }
70 return nullptr;
71 }
72 };

112 // computed defuse info for all uses and defs in the program
