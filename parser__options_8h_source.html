37// Base class for compiler options.
38// This class contains the options for the front-ends.
39// Each back-end should subclass this file.
40class ParserOptions : public Util::Options {
41 bool close_input = false;
42 static const char *defaultMessage;
43
44 // annotation names that are to be ignored by the compiler
45 std::set<cstring> disabledAnnotations;
46
47 // used to generate dump file names
48 mutable size_t dump_uid = 0;
49
50 protected:
51 // Function that is returned by getDebugHook.
52 void dumpPass(const char *manager, unsigned seq, const char *pass, const IR::Node *node) const;
53 // Checks if parsed options make sense with respect to each-other.
54 virtual void validateOptions() const;
55
56 public:
57 ParserOptions();
58 std::vector<const char *> *process(int argc, char *const argv[]) override;
59 enum class FrontendVersion { P4_14, P4_16 };
60 // Name of executable that is being run.
61 cstring exe_name;
62 // Which language to compile
63 FrontendVersion langVersion = FrontendVersion::P4_16;
64 // options to pass to preprocessor
65 cstring preprocessor_options = cstring::empty;
66 // file to compile (- for stdin)
67 std::filesystem::path file;
68 // if true preprocess only
69 bool doNotCompile = false;
70 // Compiler version.
71 cstring compilerVersion;
72 // if true skip preprocess
73 bool doNotPreprocess = false;
74 // substrings matched against pass names
75 std::vector<cstring> top4;
76 // debugging dumps of programs written in this folder
77 std::filesystem::path dumpFolder = ".";
78 // If false, optimization of callee parsers (subparsers) inlining is disabled.
79 bool optimizeParserInlining = false;
80 // Expect that the only remaining argument is the input file.
81 void setInputFile();
82 // Return target specific include path.
83 const char *getIncludePath() override;
84 // Returns the output of the preprocessor.
85 FILE *preprocess();
86 // Closes the input stream returned by preprocess.
87 void closePreprocessedInput(FILE *input) const;
88 // True if we are compiling a P4 v1.0 or v1.1 program
89 bool isv1() const;
90 // Get a debug hook function suitable for insertion
91 // in the pass managers that are executed.
92 DebugHook getDebugHook() const;
93 // Check whether this particular annotation was disabled
94 bool isAnnotationDisabled(const IR::Annotation *a) const;
95 // Search and set 'includePathOut' to be the first valid path from the
96 // list of possible relative paths.
97 bool searchForIncludePath(const char *&includePathOut, std::vector<cstring> relativePaths,
98 const char *);
101 bool noIncludes = false;
102};
